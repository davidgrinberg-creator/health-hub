        // Lógica de Alternância de Tela

        function switchScreen(screenName) {

            // Esconde todas as telas

            document.querySelectorAll('main > section').forEach(section => {

                section.classList.remove('block', 'screen-enter');

                section.classList.add('hidden');

            });

 

            // Remove 'active' de todos os botões de navegação

            document.querySelectorAll('.nav-item').forEach(nav => {

                nav.classList.remove('active');

            });

 

            // Mostra a tela selecionada

            const targetScreen = document.getElementById(`screen-${screenName}`);

            if (targetScreen) {

                targetScreen.classList.remove('hidden');

                targetScreen.classList.add('block', 'screen-enter');

            }

 

            // Ativa o botão selecionado

            const targetNav = document.getElementById(`nav-${screenName}`);

            if (targetNav) {

                targetNav.classList.add('active');

            }

 

            // Inicializa ou atualiza gráficos se for a aba de evolução

            if (screenName === 'evolution' && !window.chartsInitialized) {

                initCharts();

                window.chartsInitialized = true;

            }

        }

 

        // Lógica de Tema Dark/Light

        const themeToggleBtn = document.getElementById('themeToggle');

        const htmlElement = document.documentElement;

       

        // Verifica preferência do sistema

        if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {

            htmlElement.classList.add('dark');

        }

 

        themeToggleBtn.addEventListener('click', () => {

            htmlElement.classList.toggle('dark');

            // Atualiza os gráficos se estiverem abertos para pegar a nova cor do texto

            if (window.chartsInitialized) {

                initCharts(); // Recria os gráficos com as cores corretas

            }

        });

        function installPWA() {

            if (deferredPrompt) {

                deferredPrompt.prompt();

                deferredPrompt.userChoice.then((choiceResult) => {

                    if (choiceResult.outcome === 'accepted') {

                        document.getElementById('pwa-banner').classList.add('hidden');

                    }

                    deferredPrompt = null;

                });
